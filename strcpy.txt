char *strCp(char *str){
    int len = 0;
    while(str[len]!='\0')
        len++;
    char *newStr = malloc(len);
    int i = 0; 
    for(i=0;i<len;i++)
    {
        newStr[i]=str[i];
    }
    return newStr;
}

void main(int argc, char *argv[]) {
	char *s= strCp("sssasddsadsadsadas");
	int l = 0;
	while(s[l]!='\0')
		l++;
	printf("%s\n%d",s,l);
}
